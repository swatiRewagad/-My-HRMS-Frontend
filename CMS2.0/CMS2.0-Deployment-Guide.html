1.6 Nginx Config for Frontend

Create CMS2.0/cms-portal-frontend/nginx.conf:

server {
    listen 80;
    server_name localhost;
    root /usr/share/nginx/html;
    index index.html;

    # Angular SPA routing
    location / {
        try_files $uri $uri/ /index.html;
    }

    # API proxy to backend
    location /api/ {
        proxy_pass http://cms-backend:8082/api/;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # Cache static assets
    location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff2)$ {
        expires 1y;
        add_header Cache-Control "public, immutable";
    }

    # Security headers
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;
}

3.2 Frontend Nginx Config

Create file: CMS2.0/cms-portal-frontend/nginx.conf

# ═══════════════════════════════════════════════════════════════════
# CMS 2.0 Frontend - Nginx Configuration (inside container)
# ═══════════════════════════════════════════════════════════════════
server {
    listen 80;
    server_name _;
    root /usr/share/nginx/html;
    index index.html;

    # ── Angular SPA Routing ──
    # All routes fall back to index.html (Angular Router handles them)
    location / {
        try_files $uri $uri/ /index.html;
    }

    # ── Gzip Compression ──
    gzip on;
    gzip_vary on;
    gzip_proxied any;
    gzip_comp_level 5;
    gzip_min_length 1024;
    gzip_types
        text/plain
        text/css
        application/json
        application/javascript
        text/xml
        application/xml
        text/javascript
        image/svg+xml
        application/font-woff2;

    # ── Static Asset Caching ──
    # Angular CLI generates hashed filenames for cache-busting
    location ~* \.(js|css|woff2|woff|ttf|eot)$ {
        expires 1y;
        add_header Cache-Control "public, immutable";
        access_log off;
    }

    location ~* \.(png|jpg|jpeg|gif|ico|svg|webp)$ {
        expires 30d;
        add_header Cache-Control "public";
        access_log off;
    }

    # ── Security Headers ──
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;

    # ── Disable access to hidden files ──
    location ~ /\. {
        deny all;
        access_log off;
        log_not_found off;
    }

    # ── Health check endpoint ──
    location /healthz {
        access_log off;
        return 200 '{"status":"UP"}';
        add_header Content-Type application/json;
    }
}

3.4 MySQL Init Script

Create file: CMS2.0/cms-infra/mysql/init.sql

The docker-compose.yml mounts this file to /docker-entrypoint-initdb.d/init.sql. It runs once on first container start.

-- CMS 2.0 MySQL Init Script
-- This runs automatically on first container creation.
-- Tables are auto-created by Hibernate (ddl-auto=update), so this just sets up the user.

CREATE DATABASE IF NOT EXISTS cms_db
  CHARACTER SET utf8mb4
  COLLATE utf8mb4_unicode_ci;

-- Grant full privileges to the application user
GRANT ALL PRIVILEGES ON cms_db.* TO 'cms_user'@'%';
FLUSH PRIVILEGES;

-- Set MySQL tuning for complaint management workload
SET GLOBAL innodb_buffer_pool_size = 1073741824;  -- 1GB
SET GLOBAL max_connections = 500;
SET GLOBAL innodb_flush_log_at_trx_commit = 2;

If your Kafka cluster has authentication enabled (SASL/SCRAM or mTLS), create a KafkaUser:

# cms-kafka-user.yaml
apiVersion: kafka.strimzi.io/v1beta2
kind: KafkaUser
metadata:
  name: cms-app
  labels:
    strimzi.io/cluster: cms-kafka    # ← Your Kafka cluster name
spec:
  authentication:
    type: scram-sha-512
  authorization:
    type: simple
    acls:
    # Producer: write to complaint topics
    - resource:
        type: topic
        name: complaint
        patternType: prefix
      operations: ["Write", "Describe", "Create"]
      host: "*"
    # Producer: write to audit topic
    - resource:
        type: topic
        name: audit.events
      operations: ["Write", "Describe"]
      host: "*"
    # Consumer: read from all cms topics
    - resource:
        type: topic
        name: complaint
        patternType: prefix
      operations: ["Read", "Describe"]
      host: "*"
    - resource:
        type: topic
        name: audit.events
      operations: ["Read", "Describe"]
      host: "*"
    # Consumer group
    - resource:
        type: group
        name: cms-backend
        patternType: prefix
      operations: ["Read"]
      host: "*"

4.4.4 Backend Kafka Connection Config

Depending on your Kafka cluster's security, choose the correct configuration:

Option A: Plain (no auth) — Internal cluster, same namespace:

# In cms-config ConfigMap:
KAFKA_BOOTSTRAP=cms-kafka-kafka-bootstrap:9092

# No additional config needed. The application.yml defaults work:
# spring.kafka.bootstrap-servers: ${KAFKA_BOOTSTRAP:localhost:9092}
# spring.kafka.producer.acks: all
# spring.kafka.producer.retries: 3

Option B: Plain — Kafka in different namespace:

# Use full cluster DNS in cms-config ConfigMap:
KAFKA_BOOTSTRAP=my-cluster-kafka-bootstrap.kafka-ns.svc.cluster.local:9092

Option C: SASL/SCRAM Authentication (recommended for production):

# After creating KafkaUser 'cms-app', Strimzi creates a secret with credentials:
oc get secret cms-app -o jsonpath='{.data.password}' | base64 -d

# Add SASL properties to cms-config ConfigMap:
oc create configmap cms-kafka-config \
  --from-literal=KAFKA_BOOTSTRAP='cms-kafka-kafka-bootstrap:9093' \
  --from-literal=KAFKA_SECURITY_PROTOCOL='SASL_PLAINTEXT' \
  --from-literal=KAFKA_SASL_MECHANISM='SCRAM-SHA-512' \
  --from-literal=KAFKA_SASL_JAAS_CONFIG='org.apache.kafka.common.security.scram.ScramLoginModule required username="cms-app" password="<from-secret>";'

Option D: mTLS Authentication:

# Strimzi generates certificates in the KafkaUser secret
# Mount them as volume in backend deployment:
# Add to backend Deployment spec.containers[0]:
        volumeMounts:
        - name: kafka-certs
          mountPath: /opt/kafka/certs
          readOnly: true
      volumes:
      - name: kafka-certs
        secret:
          secretName: cms-app   # KafkaUser secret

# ConfigMap values:
KAFKA_BOOTSTRAP=cms-kafka-kafka-bootstrap:9093
KAFKA_SECURITY_PROTOCOL=SSL
KAFKA_SSL_TRUSTSTORE_LOCATION=/opt/kafka/certs/ca.p12
KAFKA_SSL_TRUSTSTORE_PASSWORD=<from cluster CA secret>
KAFKA_SSL_KEYSTORE_LOCATION=/opt/kafka/certs/user.p12
KAFKA_SSL_KEYSTORE_PASSWORD=<from user secret>

8. Troubleshooting & Common Issues

8.1 Docker Desktop Issues

ProblemCauseSolution
Backend fails to start with Communications link failure MySQL not ready when backend starts The compose has depends_on: condition: service_healthy. If still failing, increase MySQL healthcheck start_period or run docker compose restart cms-backend-1 cms-backend-2
Kafka container keeps restarting Insufficient disk space or port conflict on 29092 Run docker system prune -a --volumes to free space. Check no other process uses port 29092.
Frontend shows blank page Missing nginx.conf in cms-portal-frontend or wrong dist path Verify the Dockerfile copies from /app/dist/cms-portal-frontend/browser. Check that ng build produced output in that path.
ENOSPC: no space left on device Docker disk image full Docker Desktop → Settings → Resources → Increase disk image size. Or prune: docker system prune --all --volumes
Keycloak connection refused on port 8180 KC_HTTP_PORT not matching or still starting up Keycloak 26 takes 30-60s to start. Wait and retry. Confirm KC_HTTP_PORT: 8180 in compose.
Backend EncryptionKeyService error CMS_ENCRYPTION_SECRET not set or less than 32 chars Set in .env: CMS_ENCRYPTION_SECRET=exactly-32-characters-long-key!!
OpenSearch max virtual memory areas vm.max_map_count [65530] is too low Linux kernel limit (WSL2) In WSL terminal: sudo sysctl -w vm.max_map_count=262144. Make permanent: add to /etc/sysctl.conf
Port 80 already in use IIS, Apache, or another web server running Stop IIS: net stop w3svc or change Nginx port in compose to 8088:80

8.2 OpenShift Issues

ProblemCauseSolution
Pod CrashLoopBackOff Missing secret or configmap oc logs pod/cms-backend-xxx — look for missing env var errors. Ensure all secrets/configmaps are created first.
ImagePullBackOff Image not in registry or wrong tag oc get is to check image streams. Rebuild: oc start-build cms-backend --follow
Route returns 503 No ready pods backing the service oc get endpoints cms-backend — should list pod IPs. Check readinessProbe is passing.
PVC stuck in Pending No StorageClass or insufficient storage oc get sc to list available StorageClasses. Add storageClassName to PVC spec.
Keycloak redirect fails with invalid_redirect_uri Client redirect URIs not matching OpenShift route In Keycloak admin, update cms-frontend client's Valid Redirect URIs to https://cms.your-domain.com/*

8.3 Application-Level Issues

ProblemCauseSolution
Login works but API returns 403 User missing realm roles in Keycloak Assign correct roles (e.g., CEPC_DO, RBIO_OFFICER) to the user in Keycloak admin → Users → Role Mapping
OCR returns empty text GROQ_API_KEY invalid or PaddleOCR unreachable Test Groq: check API key at groq.com. Test Paddle: curl http://localhost:8100/health
Similar Cases not working OpenSearch index not created yet Index auto-creates on first complaint. Or manually: curl -X PUT http://localhost:9200/complaints
Kafka events not being consumed Kafka not ready when backend started, or topic doesn’t exist Topics auto-create. Restart backend: docker compose restart cms-backend-1. Check Kafka UI at port 8090.
File upload fails with 413 Nginx client_max_body_size too small The gateway nginx.conf sets client_max_body_size 50m. If you changed it, increase back to 50m.
