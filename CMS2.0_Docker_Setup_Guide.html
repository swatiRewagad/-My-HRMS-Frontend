Strategy: Infrastructure (DB, Kafka, Keycloak, OpenSearch) runs in Docker containers. Backend microservices run locally via Maven (faster dev cycle, no image rebuilding). Frontend runs via ng serve for hot-reload.

2 Start Infrastructure (Docker Containers)

2.1 Create the Docker Compose file

Create a file docker-compose-local.yml in the project root:

# File: CMS2.0/docker-compose-local.yml
# Complete infrastructure for local development

version: '3.9'

services:
  # ═══════════════════════════════════════════════
  # MySQL 8.0 (Primary Database)
  # ═══════════════════════════════════════════════
  mysql:
    image: mysql:8.0
    container_name: cms-mysql
    restart: unless-stopped
    environment:
      MYSQL_ROOT_PASSWORD: root
      MYSQL_DATABASE: cms_db
    ports:
      - "3306:3306"
    volumes:
      - mysql_data:/var/lib/mysql
    command: >
      --character-set-server=utf8mb4
      --collation-server=utf8mb4_unicode_ci
      --max-connections=500
    healthcheck:
      test: ["CMD", "mysqladmin", "ping", "-h", "localhost", "-u", "root", "-proot"]
      interval: 10s
      timeout: 5s
      retries: 5
    networks:
      - cms-net

  # ═══════════════════════════════════════════════
  # Apache Kafka 3.7 (KRaft mode - No Zookeeper)
  # ═══════════════════════════════════════════════
  kafka:
    image: apache/kafka:3.7.0
    container_name: cms-kafka
    restart: unless-stopped
    environment:
      KAFKA_NODE_ID: 1
      KAFKA_PROCESS_ROLES: broker,controller
      KAFKA_LISTENERS: PLAINTEXT://0.0.0.0:9092,CONTROLLER://0.0.0.0:9093,EXTERNAL://0.0.0.0:29092
      KAFKA_ADVERTISED_LISTENERS: PLAINTEXT://kafka:9092,EXTERNAL://localhost:29092
      KAFKA_CONTROLLER_LISTENER_NAMES: CONTROLLER
      KAFKA_LISTENER_SECURITY_PROTOCOL_MAP: CONTROLLER:PLAINTEXT,PLAINTEXT:PLAINTEXT,EXTERNAL:PLAINTEXT
      KAFKA_CONTROLLER_QUORUM_VOTERS: 1@kafka:9093
      KAFKA_OFFSETS_TOPIC_REPLICATION_FACTOR: 1
      KAFKA_TRANSACTION_STATE_LOG_REPLICATION_FACTOR: 1
      KAFKA_TRANSACTION_STATE_LOG_MIN_ISR: 1
      KAFKA_LOG_DIRS: /var/lib/kafka/data
      CLUSTER_ID: 'cms-kafka-local-001'
    ports:
      - "29092:29092"
    volumes:
      - kafka_data:/var/lib/kafka/data
    healthcheck:
      test: ["CMD-SHELL", "/opt/kafka/bin/kafka-topics.sh --bootstrap-server localhost:9092 --list || exit 1"]
      interval: 15s
      timeout: 10s
      retries: 10
      start_period: 30s
    networks:
      - cms-net

  # ═══════════════════════════════════════════════
  # Keycloak 26.0 (Identity & Access Management)
  # ═══════════════════════════════════════════════
  keycloak:
    image: quay.io/keycloak/keycloak:26.0
    container_name: cms-keycloak
    restart: unless-stopped
    environment:
      KEYCLOAK_ADMIN: admin
      KEYCLOAK_ADMIN_PASSWORD: admin
      KC_DB: postgres
      KC_DB_URL: jdbc:postgresql://keycloak-db:5432/keycloak
      KC_DB_USERNAME: keycloak
      KC_DB_PASSWORD: keycloak
      KC_HTTP_PORT: 8180
      KC_HOSTNAME_STRICT: "false"
      KC_PROXY_HEADERS: xforwarded
    command: start-dev
    ports:
      - "8180:8180"
    depends_on:
      keycloak-db:
        condition: service_healthy
    networks:
      - cms-net

  keycloak-db:
    image: postgres:16-alpine
    container_name: cms-keycloak-db
    restart: unless-stopped
    environment:
      POSTGRES_DB: keycloak
      POSTGRES_USER: keycloak
      POSTGRES_PASSWORD: keycloak
    volumes:
      - keycloak_db_data:/var/lib/postgresql/data
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U keycloak"]
      interval: 5s
      timeout: 3s
      retries: 10
    networks:
      - cms-net

  # ═══════════════════════════════════════════════
  # OpenSearch 2.18 (Full-Text Search)
  # ═══════════════════════════════════════════════
  opensearch:
    image: opensearchproject/opensearch:2.18.0
    container_name: cms-opensearch
    restart: unless-stopped
    environment:
      discovery.type: single-node
      DISABLE_SECURITY_PLUGIN: "true"
      OPENSEARCH_JAVA_OPTS: "-Xms512m -Xmx512m"
    ports:
      - "9200:9200"
    volumes:
      - opensearch_data:/usr/share/opensearch/data
    healthcheck:
      test: ["CMD-SHELL", "curl -sf http://localhost:9200/_cluster/health || exit 1"]
      interval: 15s
      timeout: 10s
      retries: 10
    networks:
      - cms-net

  # ═══════════════════════════════════════════════
  # Kafka UI (Management Console)
  # ═══════════════════════════════════════════════
  kafka-ui:
    image: provectuslabs/kafka-ui:latest
    container_name: cms-kafka-ui
    restart: unless-stopped
    environment:
      KAFKA_CLUSTERS_0_NAME: cms-local
      KAFKA_CLUSTERS_0_BOOTSTRAPSERVERS: kafka:9092
    ports:
      - "8090:8080"
    depends_on:
      kafka:
        condition: service_healthy
    networks:
      - cms-net

  # ═══════════════════════════════════════════════
  # Prometheus (Metrics Collection)
  # ═══════════════════════════════════════════════
  prometheus:
    image: prom/prometheus:v2.54.0
    container_name: cms-prometheus
    restart: unless-stopped
    volumes:
      - ./deployment/docker/prometheus.yml:/etc/prometheus/prometheus.yml
    ports:
      - "9090:9090"
    extra_hosts:
      - "host.docker.internal:host-gateway"
    networks:
      - cms-net

  # ═══════════════════════════════════════════════
  # Grafana (Dashboards & Monitoring)
  # ═══════════════════════════════════════════════
  grafana:
    image: grafana/grafana:11.3.0
    container_name: cms-grafana
    restart: unless-stopped
    environment:
      GF_SECURITY_ADMIN_USER: admin
      GF_SECURITY_ADMIN_PASSWORD: admin
    ports:
      - "3000:3000"
    volumes:
      - grafana_data:/var/lib/grafana
    depends_on:
      - prometheus
    networks:
      - cms-net

volumes:
  mysql_data:
  kafka_data:
  keycloak_db_data:
  opensearch_data:
  grafana_data:

networks:
  cms-net:
    driver: bridge

# From project root (CMS2.0/)
docker compose -f docker-compose-local.yml up -d

3.1 Access Keycloak Admin Console

# Open in browser:
http://localhost:8180

# Login:
Username: admin
Password: admin

3.2 Create Realm

  1. Click dropdown (top-left, shows "master") → Create realm
  2. Realm name: cms
  3. Click Create

3.3 Create Client

  1. Go to Clients → Create client
  2. Client ID: cms-frontend
  3. Client authentication: OFF (public client for SPA)
  4. Authorization: OFF
  5. Click Next
  6. Valid redirect URIs: http://localhost:4200/*
  7. Valid post logout redirect URIs: http://localhost:4200/*
  8. Web origins: http://localhost:4200
  9. Click Save

4 Build & Start Backend Services

4.1 Update Application Configs for Docker

The backend services need to connect to Docker containers. Key config changes:

cms-backend/src/main/resources/application.yml

# Database - points to Docker MySQL
spring:
  datasource:
    url: jdbc:mysql://localhost:3306/cms_db?useSSL=false&allowPublicKeyRetrieval=true&serverTimezone=UTC
    username: root
    password: root

# Kafka - points to Docker Kafka (EXTERNAL listener)
  kafka:
    bootstrap-servers: localhost:29092

# Keycloak Admin
keycloak:
  admin:
    server-url: http://localhost:8180
    realm: cms
    client-id: admin-cli
    username: admin
    password: admin

cms-api-gateway/src/main/resources/application.yml

spring:
  security:
    oauth2:
      resourceserver:
        jwt:
          issuer-uri: http://localhost:8180/realms/cms
          jwk-set-uri: http://localhost:8180/realms/cms/protocol/openid-connect/certs

All microservices (dev-local profile) - Kafka

# In each service's application-dev-local.yml or application.yml:
spring:
  kafka:
    bootstrap-servers: localhost:29092

4.2 Build All Modules

# From project root (CMS2.0/)
mvn clean install -DskipTests

# This builds all 14 modules. Takes ~2-3 minutes first time.

4.3 Start Services (Open separate terminals for each)

Order matters! Start in this sequence. Wait for each to show "Started" before next.

Terminal 1: API Gateway (Port 8080)

cd cms-api-gateway
mvn spring-boot:run -Dspring-boot.run.profiles=dev-local -DskipTests

Terminal 2: Backend (Port 8082) — MOST IMPORTANT

cd cms-backend
mvn spring-boot:run -DskipTests

Terminal 3: Workflow Service (Port 8083)

cd cms-workflow-service
mvn spring-boot:run -Dspring-boot.run.profiles=dev-local -DskipTests

Terminal 4: Eligibility Service (Port 8081)

cd cms-eligibility-service

Terminal 5: Rules Service (Port 8084)

cd cms-rules-service

4.4 Quick Start Script (All Services)

Create start-all.bat (Windows) or start-all.sh (Mac/Linux):

Windows: start-all.bat

@echo off
echo Starting CMS 2.0 Backend Services...

start "API-Gateway" cmd /k "cd cms-api-gateway && mvn spring-boot:run -Dspring-boot.run.profiles=dev-local -DskipTests"
timeout /t 15

start "Backend" cmd /k "cd cms-backend && mvn spring-boot:run -DskipTests"
timeout /t 20

start "Workflow" cmd /k "cd cms-workflow-service && mvn spring-boot:run -Dspring-boot.run.profiles=dev-local -DskipTests"
start "Eligibility" cmd /k "cd cms-eligibility-service && mvn spring-boot:run -Dspring-boot.run.profiles=dev-local -DskipTests"
start "Rules" cmd /k "cd cms-rules-service && mvn spring-boot:run -Dspring-boot.run.profiles=dev-local -DskipTests"

echo All services starting. Check individual windows for status.

Mac/Linux: start-all.sh

#!/bin/bash
echo "Starting CMS 2.0 Backend Services..."

cd cms-api-gateway && mvn spring-boot:run -Dspring-boot.run.profiles=dev-local -DskipTests &
sleep 15

cd ../cms-backend && mvn spring-boot:run -DskipTests &

! Troubleshooting

Container won't start

# Check logs
docker compose -f docker-compose-local.yml logs mysql
docker compose -f docker-compose-local.yml logs keycloak
docker compose -f docker-compose-local.yml logs kafka

# Restart a specific service
docker compose -f docker-compose-local.yml restart keycloak

# Nuclear option - start fresh
docker compose -f docker-compose-local.yml down -v
docker compose -f docker-compose-local.yml up -d

Backend can't connect to MySQL

# Verify MySQL is accessible from host
docker exec cms-mysql mysql -u root -proot -e "SHOW DATABASES;"

# If port conflict, check what's using 3306
netstat -ano | findstr :3306    # Windows
lsof -i :3306                   # Mac/Linux

Keycloak login fails ("Invalid redirect URI")

# Ensure client redirect URIs match exactly
# In Keycloak Admin → Clients → cms-frontend → Settings:
# Valid Redirect URIs: http://localhost:4200/*
# Web Origins: http://localhost:4200

Kafka connection refused

# The EXTERNAL listener on port 29092 is for host → container communication
# Make sure your application.yml uses:
spring.kafka.bootstrap-servers: localhost:29092

# NOT localhost:9092 (that's internal to Docker network)

Frontend "Keycloak init failed"

# Check environment.ts has correct URL:
keycloakUrl: 'http://localhost:8180'  # NOT 8080!

# Verify Keycloak realm exists:
curl http://localhost:8180/realms/cms

"Send for Approval" not working

# Backend needs to reach Keycloak to fetch users
# In cms-backend/src/main/resources/application.yml:
keycloak:
  admin:
    server-url: http://localhost:8180   # Must match Docker Keycloak port

Stop Everything

# Stop Docker infrastructure
docker compose -f docker-compose-local.yml down

# Stop backend services (find Java processes)
# Windows:
taskkill /f /im java.exe

# Mac/Linux:
pkill -f "spring-boot:run"

# Stop frontend
# Press Ctrl+C in the ng serve terminal

Reset Everything (Fresh Start)

# Remove all Docker volumes (DELETES ALL DATA)
docker compose -f docker-compose-local.yml down -v

# Remove Maven builds
mvn clean

# Start over from Step 2
docker compose -f docker-compose-local.yml up -d
