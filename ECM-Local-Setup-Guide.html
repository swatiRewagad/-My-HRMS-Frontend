DB Complete Database Script

Note: Spring Boot ddl-auto: update auto-creates all tables on startup. This script is for manual setup or reference only.

Create Database

CREATE DATABASE IF NOT EXISTS ecm_db CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE ecm_db;

Table: ECM_USERS

CREATE TABLE ECM_USERS (
    id              BIGINT AUTO_INCREMENT PRIMARY KEY,
    username        VARCHAR(100) NOT NULL UNIQUE,
    display_name    VARCHAR(200) NOT NULL,
    email           VARCHAR(200),
    role            VARCHAR(50),
    initials        VARCHAR(10),
    created_at      DATETIME
);

Table: PROJECTS

CREATE TABLE PROJECTS (
    id              BIGINT AUTO_INCREMENT PRIMARY KEY,
    code            VARCHAR(50) NOT NULL UNIQUE,
    name            VARCHAR(200) NOT NULL,
    description     VARCHAR(500),
    status          VARCHAR(20) NOT NULL DEFAULT 'active',
    created_by      BIGINT,
    created_at      DATETIME,
    updated_at      DATETIME
);

Table: FOLDERS

CREATE TABLE FOLDERS (
    id              BIGINT AUTO_INCREMENT PRIMARY KEY,
    name            VARCHAR(300) NOT NULL,
    visibility      VARCHAR(20) NOT NULL,
    description     VARCHAR(1000),
    parent_id       BIGINT,
    owner_id        BIGINT NOT NULL,
    path            VARCHAR(500),
    created_at      DATETIME,
    updated_at      DATETIME,
    FOREIGN KEY (parent_id) REFERENCES FOLDERS(id),
    FOREIGN KEY (owner_id) REFERENCES ECM_USERS(id)
);

Table: FILES

CREATE TABLE FILES (
    id              BIGINT AUTO_INCREMENT PRIMARY KEY,
    name            VARCHAR(500) NOT NULL,
    original_name   VARCHAR(500) NOT NULL,
    content_type    VARCHAR(100),
    size            BIGINT,
    storage_path    VARCHAR(1000),
    folder_id       BIGINT NOT NULL,
    uploaded_by     BIGINT NOT NULL,
    status          VARCHAR(50) DEFAULT 'active',
    uploaded_at     DATETIME,
    updated_at      DATETIME,
    FOREIGN KEY (folder_id) REFERENCES FOLDERS(id),
    FOREIGN KEY (uploaded_by) REFERENCES ECM_USERS(id)
);

Table: FOLDER_ACCESS

CREATE TABLE FOLDER_ACCESS (
    id              BIGINT AUTO_INCREMENT PRIMARY KEY,
    folder_id       BIGINT NOT NULL,
    user_id         BIGINT NOT NULL,
    permission      VARCHAR(20) NOT NULL,
    granted_at      DATETIME,
    FOREIGN KEY (folder_id) REFERENCES FOLDERS(id),
    FOREIGN KEY (user_id) REFERENCES ECM_USERS(id)
);

Table: FILE_SHARES

CREATE TABLE FILE_SHARES (
    id              BIGINT AUTO_INCREMENT PRIMARY KEY,
    file_id         BIGINT NOT NULL,
    shared_by       BIGINT NOT NULL,
    shared_with     BIGINT,
    share_type      VARCHAR(20) NOT NULL,
    share_token     VARCHAR(100) UNIQUE,
    permission      VARCHAR(20),
    expires_at      DATETIME,
    shared_at       DATETIME,
    FOREIGN KEY (file_id) REFERENCES FILES(id),
    FOREIGN KEY (shared_by) REFERENCES ECM_USERS(id),
    FOREIGN KEY (shared_with) REFERENCES ECM_USERS(id)
);

Table: ECM_ACTIVITY_LOG

CREATE TABLE ECM_ACTIVITY_LOG (
    id              BIGINT AUTO_INCREMENT PRIMARY KEY,
    action          VARCHAR(50) NOT NULL,
    entity_type     VARCHAR(20) NOT NULL,
    entity_name     VARCHAR(500),
    user_id         BIGINT,
    user_name       VARCHAR(200),
    performed_at    DATETIME,
    FOREIGN KEY (user_id) REFERENCES ECM_USERS(id)
);

Table: CHUNK_UPLOADS

CREATE TABLE CHUNK_UPLOADS (
    id              BIGINT AUTO_INCREMENT PRIMARY KEY,
    upload_id       VARCHAR(100) NOT NULL UNIQUE,
    file_name       VARCHAR(500) NOT NULL,
    content_type    VARCHAR(100),
    total_size      BIGINT,
    total_chunks    INT,
    chunks_received INT DEFAULT 0,
    folder_id       BIGINT NOT NULL,
    uploaded_by     BIGINT NOT NULL,
    status          VARCHAR(20) DEFAULT 'uploading',
    created_at      DATETIME,
    updated_at      DATETIME,
    FOREIGN KEY (folder_id) REFERENCES FOLDERS(id),
    FOREIGN KEY (uploaded_by) REFERENCES ECM_USERS(id)
);

Table: INVOICE_EXTRACTIONS

CREATE TABLE INVOICE_EXTRACTIONS (
    id              BIGINT AUTO_INCREMENT PRIMARY KEY,
    file_id         BIGINT NOT NULL,
    invoice_number  VARCHAR(200),
    invoice_date    VARCHAR(200),
    due_date        VARCHAR(200),
    vendor_name     VARCHAR(500),
    vendor_address  VARCHAR(500),
    customer_name   VARCHAR(500),
    customer_address VARCHAR(500),
    subtotal        VARCHAR(100),
    tax             VARCHAR(100),
    total_amount    VARCHAR(100),
    currency        VARCHAR(100),
    line_items_json TEXT,
    raw_text        TEXT,
    status          VARCHAR(50) DEFAULT 'completed',
    error_message   VARCHAR(1000),
    extracted_at    DATETIME,
    FOREIGN KEY (file_id) REFERENCES FILES(id)
);

Table: PROJECT_UPLOAD_CONFIGS

CREATE TABLE PROJECT_UPLOAD_CONFIGS (
    id                          BIGINT AUTO_INCREMENT PRIMARY KEY,
    project_id                  BIGINT NOT NULL UNIQUE,
    max_file_size_bytes         BIGINT NOT NULL,
    total_allocated_storage_bytes BIGINT NOT NULL,
    allowed_content_types       VARCHAR(500) NOT NULL,
    upload_base_path            VARCHAR(1000) NOT NULL,
    created_at                  DATETIME,
    updated_at                  DATETIME,
    FOREIGN KEY (project_id) REFERENCES PROJECTS(id)
);

Table: DOCUMENT_TYPE_CONFIGS

CREATE TABLE DOCUMENT_TYPE_CONFIGS (
    id                  BIGINT AUTO_INCREMENT PRIMARY KEY,
    project_id          BIGINT NOT NULL,
    type_name           VARCHAR(100) NOT NULL,
    type_code           VARCHAR(50) NOT NULL,
    description         VARCHAR(500),
    extraction_enabled  BOOLEAN NOT NULL DEFAULT FALSE,
    status              VARCHAR(20) NOT NULL DEFAULT 'active',
    created_at          DATETIME,
    updated_at          DATETIME,
    FOREIGN KEY (project_id) REFERENCES PROJECTS(id)
);

Table: INVOICE_FIELD_CONFIGS

CREATE TABLE INVOICE_FIELD_CONFIGS (
    id                  BIGINT AUTO_INCREMENT PRIMARY KEY,
    doc_type_config_id  BIGINT NOT NULL,
    field_name          VARCHAR(100) NOT NULL,
    field_key           VARCHAR(100) NOT NULL,
    field_type          VARCHAR(50) NOT NULL,
    required            BOOLEAN NOT NULL DEFAULT FALSE,
    display_order       INT NOT NULL DEFAULT 0,
    field_category      VARCHAR(50) NOT NULL DEFAULT 'header',
    created_at          DATETIME,
    FOREIGN KEY (doc_type_config_id) REFERENCES DOCUMENT_TYPE_CONFIGS(id)
);

Tip: You don't need to run these scripts manually. Spring Boot auto-creates tables and the DataInitializer class seeds sample data on first run. Use these scripts only if you need to reset or recreate the database manually.

1 Clone the Repository

git clone https://github.com/swatiRewagad/-My-HRMS-Frontend.git
cd My-HRMS-Frontend

2 MySQL Database Setup

  1. Start MySQL service
  2. Login to MySQL: 
    mysql -u root -p
  3. Create the database (auto-created by Spring Boot, but you can do it manually): 
    CREATE DATABASE IF NOT EXISTS ecm_db;
  4. Verify: 
    SHOW DATABASES;
Note: Default credentials are root / root. If your MySQL password is different, set the DB_PASSWORD environment variable.

3 ECM Backend Setup

3.1 Navigate to backend directory

cd ecm-backend

3.2 Set environment variables (if needed)

Variable Default Description
DB_PASSWORD root MySQL root password
ECM_STORAGE_PATH C:/ecm-storage Local folder for file storage
LLM_API_KEY (empty) Optional: Anthropic API key for AI features

Windows (PowerShell):

$env:DB_PASSWORD = "root"
$env:ECM_STORAGE_PATH = "C:/ecm-storage"

Windows (CMD):

set DB_PASSWORD=root
set ECM_STORAGE_PATH=C:/ecm-storage

3.3 Create storage directory

mkdir C:\ecm-storage

3.4 Build and Run

mvn clean install -DskipTests
mvn spring-boot:run

3.5 Verify

  • Backend starts on: http://localhost:8081
  • Health check: Open http://localhost:8081/api/files in browser (should return JSON)

4 ECM Frontend Setup

4.1 Navigate to frontend directory

cd ecm-frontend

4.2 Install dependencies

npm install

4.3 Start development server

npm start

4.4 Verify

  • Frontend starts on: http://localhost:4200
  • Open http://localhost:4200 in browser

5 Verify End-to-End

  1. Open http://localhost:4200 in Chrome/Edge
  2. You should see the ECM file manager interface
  3. Try uploading a file — it should save to C:/ecm-storage
  4. Try creating folders, sharing links, previewing documents

Troubleshooting

MySQL connection refused

  • Ensure MySQL service is running: net start MySQL80 (Windows)
  • Check port 3306 is not blocked
  • Verify credentials: mysql -u root -proot

Port 8081 already in use

  • Kill the process: netstat -ano | findstr :8081 then taskkill /F /PID <pid>
  • Or change port in ecm-backend/src/main/resources/application.yml

Port 4200 already in use

  • Run on different port: ng serve --port 4201

npm install fails

  • Clear cache: npm cache clean --force
  • Delete node_modules and retry: rm -rf node_modules && npm install

Backend build fails (Java version)

  • Ensure Java 21 is installed: java -version
  • Set JAVA_HOME to JDK 21 path

File upload fails

  • Ensure C:/ecm-storage directory exists and is writable
  • Check backend logs for errors
